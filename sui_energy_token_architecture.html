Sui Move is object-centric. Coin<ENERGY> are owned objects — parallel execution, no consensus required. EnergyMeter is a shared object — consensus-serialised writes, one per physical device.

// ── Module: energy_grid::energy_token ───────────────────────────
module energy_grid::energy_token {
    use sui::coin::{Coin, TreasuryCap};
    use sui::object::UID;

    // Fungible token witness — one-time-use, drop-only ────────────
    public struct ENERGY has drop {}

    // Non-fungible batch receipt (owned by producer) ─────────────
    public struct EnergyBatch has key, store {
        id:         UID,
        producer:   address,
        kwh:        u64,          // micro-units: 1_000_000 = 1 kWh
        source:     u8,           // 0=solar 1=wind 2=hydro 3=grid
        timestamp:  u64,          // epoch_timestamp_ms at mint
        meter_id:   ID,           // links to EnergyMeter shared obj
        oracle_sig: vector<u8>,   // raw ECDSA sig from IoT device
        redeemed:   bool,
    }

    // Shared meter state — one per physical device ───────────────
    public struct EnergyMeter has key {
        id:              UID,
        device_pubkey:   vector<u8>, // secp256k1 device pubkey
        total_kwh:       u64,
        last_reading_ts: u64,
        is_certified:    bool,
        producer_addr:   address,
    }

    // Module init: create currency, transfer TreasuryCap ─────────
    fun init(witness: ENERGY, ctx: &mut TxContext) {
        let (treasury, metadata) = coin::create_currency(
            witness,
            6,               // decimals: 1 ENERGY = 1 kWh
            b"ENERGY",
            b"Energy Token",
            b"Tokenised kWh on Sui",
            option::none(), ctx,
        );
        transfer::public_freeze_object(metadata);
        transfer::public_transfer(treasury, tx_context::sender(ctx));
    }
}
// ── Module: minting_engine ───────────────────────────────────────
public fun mint_energy(
    cap:        &mut TreasuryCap<ENERGY>,
    meter:      &mut EnergyMeter,     // shared obj — consensus write
    kwh:        u64,
    oracle_sig: vector<u8>,
    ctx:        &mut TxContext,
): Coin<ENERGY> {
    // 1. On-chain sig verification over (meter_id ‖ kwh ‖ timestamp)
    assert!(verify_sig(&oracle_sig, meter, kwh), EInvalidAttestation);
    // 2. Replay-protection: timestamp must advance past last reading
    assert!(is_fresh(meter, ctx), EStaleReading);
    // 3. Update shared meter state
    meter.total_kwh = meter.total_kwh + kwh;
    meter.last_reading_ts = tx_context::epoch_timestamp_ms(ctx);
    // 4. Mint: 1 kWh = 1_000_000 base units (6 decimals)
    coin::mint(cap, kwh * 1_000_000, ctx)
}

// ── Module: compliance_registry ──────────────────────────────────
public struct KycRecord has store {
    status:       u8,   // 0=pending 1=cleared 2=suspended
    jurisdiction: u8,   // ISO country code
    expiry:       u64,  // epoch ms
}

public fun assert_cleared(reg: &ComplianceRegistry, addr: address) {
    let rec = table::borrow(&reg.records, addr);
    assert!(rec.status == 1, ENotKYCCleared);
    assert!(rec.expiry > tx_context::epoch_timestamp_ms(ctx), EKYCExpired);
}
